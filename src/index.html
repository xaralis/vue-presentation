Direktivy

v-bind
Binding HTML atributů
nastaví atribut v DOM 

<!-- plná syntaxe -->
<a v-bind:href="url"></a>
<!-- zkratka -->
<a :href="url"></a>
v-on
Binding DOM eventů
při triggeru eventu zavolá VM funkci 

<!-- plná syntaxe -->
<a v-on:click="doMethod"></a>
<!-- zkratka -->
<a @click="doMethod"></a>
Modifikátory

Vypočítané proměnné a watchery

Watch

Podobně jako v Angularu, při změně proměnné zavolá funkci 

var vm = new Vue({
  el: '#demo',
  data: {
    firstName: 'Foo',
    lastName: 'Bar',
    fullName: 'Foo Bar'
  },
  watch: {
    firstName: function (val) {
      this.fullName = val + ' ' + this.lastName
    },
    lastName: function (val) {
      this.fullName = this.firstName + ' ' + val
    }
  }
})

Vypočítané proměnné

Deklarace proměnné na základě její závislosti na jiných proměnných, automatická reaktivita. Preferované řešení pro odvozené proměnné. 

<div id="demo">{{ fullName }}</div>
                

var vm = new Vue({
  el: '#demo',
  data: {
    firstName: 'Foo',
    lastName: 'Bar'
  },
  computed: {
    fullName: function () {
      return this.firstName + ' ' + this.lastName
    }
  }
})

Settery vypočítaných proměnných

Vypočítaná proměnná může mít i setter, který se zavolá, když do ni někdo zkusí něco přiřadit. 

//...
computed: {
  fullName: {
    // getter
    get: function () {
      return this.firstName + ' ' + this.lastName
    },
    // setter
    set: function (newValue) {
      var names = newValue.split(' ')
      this.firstName = names[0]
      this.lastName = names[names.length - 1]
    }
  }
}
//...

Modifikátory kláves


<!-- propagace click eventu se zastaví automaticky -->
<a @click.stop="oThis"></a>
<!-- submit event nezpůsobí reload stránky -->
<form @submit.prevent="onSubmit"></form>
<!-- modifikátory lze řetězit -->
<a @click.stop.prevent="doThat"></a>
<!-- k dispozici je mnoho dalších modifikátorů: .tab, .delete, .esc, .space, ... -->
<input @keyup.enter="submit">

Základní struktura komponenty


export default {
    name: 'RangeSlider', // název
    mixins: [], // mixiny umožňují sdílet funkcionalitu
    props: {}, // vstupní proměnné, lze definovat i typovou kontrolu
    data() {}, // lokální proměnné komponenty
    computed: {}, // reaktivní vypočítané proměnné
    components: {}, // registrace child komponent, které tato komponenta využívá
    watch: {}, // definice watcherů
    methods: {}, // definice metod

    // základní lifecycle hooky
    beforeCreate() {}, // před vytvořením komponenty
    created() {}, // po tom co je komponenta vytvořena
    beforeMount() {}, // před připojením komponenty na DOM element
    mounted() {}, // po připojení na DOM element
    beforeUpdate() {}, // před aktualizací datového modelu
    updated() {},   // po aktualizaci datového modelu
    beforeDestroy() {}, // před odstraněním komponenty z DOMu
    destroyed() {}, // po odstranění komponenty z DOMu
    errorCaptured() {},
    serverPrefetch() {}, // pro načtení async dat při SSR
};

vue-router

  • Zpřístupňuje navigaci pro SPA aplikace
  • Perfektní integrace, de-facto routing řešení vyvíjené core týmem

Základní konfigurace


// Komponenty
const Foo = { template: '<div>foo</div>' }
const User = { template: '<div>user</div>' }

// Jednotlivé routy
const routes = [
  { path: '/foo', name: 'foo', component: Foo },
  { path: '/user/:userId', name: 'user', component: User }
]

// Instance routeru
const router = new VueRouter({ routes })

// Root instance Vue
const app = new Vue({ router }).$mount('#app')

Navigace


// bez parametrů
router.push('foo')
// s parametry
router.push({ name: 'user', params: { userId: '123' } })
// s query stringem: /foo?param=value
router.push({ name: 'foo', query: { param: 'value' } })
